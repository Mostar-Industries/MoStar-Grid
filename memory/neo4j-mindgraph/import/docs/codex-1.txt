class MostarRegistry:

@classmethod

def load_registry(cls) -> Dict[str, Dict]:

if cls.TRUSTED_DATABASE.exists():

return json.loads(cls.TRUSTED_DATABASE.read_text())

return {}



@classmethod

def save_registry(cls, data: Dict[str, Dict]):

cls.TRUSTED_DATABASE.write_text(json.dumps(data, indent=2))



@classmethod

registry = cls.load_registry()

class MostarRegistry:

    def load_registry(cls) -> Dict[str, Dict]:
        if cls.TRUSTED_DATABASE.exists():
            return json.loads(cls.TRUSTED_DATABASE.read_text())
        return {}

    @classmethod
    def save_registry(cls, data: Dict[str, Dict]):
        cls.TRUSTED_DATABASE.write_text(json.dumps(data, indent=2))
