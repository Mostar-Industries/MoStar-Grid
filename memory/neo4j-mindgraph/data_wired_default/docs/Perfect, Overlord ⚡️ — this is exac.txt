import json

# Helper to read files based on extension
def read_file_content(file_path):
    try:
        if file_path.endswith(".json"):
            with open(file_path, "r", encoding="utf-8") as f:
                return json.load(f)
        elif file_path.endswith(".yaml") or file_path.endswith(".yml"):
            import yaml
            with open(file_path, "r", encoding="utf-8") as f:
                return yaml.safe_load(f)
        elif file_path.endswith(".md") or file_path.endswith(".txt") or file_path.endswith(".ts"):
            with open(file_path, "r", encoding="utf-8") as f:
                return f.read()
        else:
            return f"Unsupported file type: {file_path}"
    except Exception as e:
        return str(e)

# Draw nodes with colors by type
colors = []
for node in G.nodes(data=True):
    if node[1]["type"] == "bundle":
        colors.append("gold")
    elif node[1]["type"] == "good":
        colors.append("green")
    else:
        colors.append("red")

plt.figure(figsize=(14, 10))
nx.draw_networkx_nodes(G, pos, node_color=colors, node_size=1500, alpha=0.85)
nx.draw_networkx_labels(G, pos, font_size=9, font_weight="bold")
