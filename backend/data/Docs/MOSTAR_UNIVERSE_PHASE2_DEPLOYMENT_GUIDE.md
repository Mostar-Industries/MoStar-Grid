import json
import random
from pathlib import Path

class IbibioInterface:
    """Native Ibibio language interface for REMOSTAR DCX001."""
    
    def __init__(self):
        self.dictionary_path = Path("/home/claude/mostar_excavation/ibibio_language_system/ibibio_database/ibibio_dictionary.json")
        self.audio_base_url = "http://localhost:8000/static/audio/ibibio/"
        self.load_dictionary()
    
    def load_dictionary(self):
        """Load Ibibio dictionary."""
        with open(self.dictionary_path, 'r', encoding='utf-8') as f:
            data = json.load(f)
            self.words = {entry['ibibio']: entry for entry in data}
    
    def translate_to_ibibio(self, english_word: str) -> dict:
        """Translate English word to Ibibio."""
        for ibibio, entry in self.words.items():
            if entry['english'].lower() == english_word.lower():
                return {
                    'ibibio': ibibio,
                    'tone_pattern': entry.get('tone_pattern'),
                    'audio_url': f"{self.audio_base_url}{entry.get('audio_file')}"
                }
        return None
    
    def greet_in_ibibio(self) -> dict:
        """Return native Ibibio greeting."""
        greetings = [
            {'text': 'Nnọ', 'meaning': 'Welcome', 'audio': 'greeting_nno.mp3'},
            {'text': 'Mmo', 'meaning': 'Hello', 'audio': 'greeting_mmo.mp3'},
        ]
        greeting = random.choice(greetings)
        greeting['audio_url'] = f"{self.audio_base_url}{greeting['audio']}"
        return greeting
    
    def affirm_in_ibibio(self, affirmative: bool = True) -> dict:
        """Return Ibibio affirmation or negation."""
        if affirmative:
            return {
                'text': 'Afo',
                'meaning': 'Yes',
                'audio_url': f"{self.audio_base_url}response_afo.mp3"

        else:
            return {
                'text': 'Mme',
                'meaning': 'No',
                'audio_url': f"{self.audio_base_url}response_mme.mp3"

    def thank_in_ibibio(self) -> dict:

from neo4j import GraphDatabase

class EntityAwareRouter:
    """Route API calls to appropriate entities based on Neo4j ownership."""
    
    def __init__(self, neo4j_uri, neo4j_user, neo4j_password):
        self.driver = GraphDatabase.driver(neo4j_uri, auth=(neo4j_user, neo4j_password))
    
    def get_endpoint_owner(self, endpoint_path: str) -> str:
        """Query Neo4j to find entity owner of endpoint."""
        with self.driver.session() as session:
            result = session.run(
                """
                MATCH (a:APIEndpoint {path: $path})<-[:EXPOSES_API]-(e:Entity)
                RETURN e.name AS owner
                """,
                path=endpoint_path
            )
            record = result.single()
            return record['owner'] if record else None
    
    def route_to_entity(self, endpoint_path: str, request_data: dict) -> dict:
        """Route request to entity based on ownership."""
        owner = self.get_endpoint_owner(endpoint_path)
        
        if owner == "Mo":
            return self.route_to_mo(request_data)
        elif owner == "DeepCAL":
            return self.route_to_deepcal(request_data)
        elif owner == "RAD-X-FLB":
            return self.route_to_radx(request_data)
        # ... etc
        
        return {"error": "Unknown entity owner"}
